13.1 First strata

str = "smk"
str.lab = 1
dataset1 <- dataset[dataset[[str]] == str.lab,]
M <- table(dataset1[[ex]], dataset1[[outcome]])
M3 <- matrix(c(M[2,2],M[2,1],M[1,2],M[1,1]), nrow = 2, byrow = TRUE)
res1 <- epi.2by2(dat = M3, method = "cross.sectional",
         conf.level = 0.95, units = 1, interpret = FALSE, outcome = "as.columns")
res1
